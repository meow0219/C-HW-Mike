#include <stdio.h>
#include <stdlib.h>
#include <string.h>

long long convert(int);

int main()
{
    int input;   
    printf("輸入一個十進位數: ");
    scanf_s("%d", &input);
    printf("十進位數 %d 轉換為二進位數: %lld \n", input, convert(input));
    
    return 0;
}

long long convert(int n)
{
    long long binaryNumber = 0;
    int remainder, i = 1;

    while (n != 0)
    {
        remainder = n % 2;
        n /= 2;
        binaryNumber += remainder * i;
        i *= 10;
    }
    return binaryNumber;
}
