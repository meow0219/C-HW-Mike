#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void d2b(int, char*);

int main()
{
    int input;
    char str[32];    
    printf("輸入一個十進位數: ");
    scanf_s("%d", &input);
    d2b(input, str);
    printf("十進位數 %d 轉換為二進位數: %s \n", input, str);
    
    return 0;
}

void d2b(int n,char* pbin)
{
    long long binaryNumber = 0;
    int i = 0;
    int j = 0;
    char result[32];
    while (n != 0) {
        binaryNumber = n % 2;
        n /= 2;
        result[i] = binaryNumber+'0';        
        i++;
    }
    for (j = 0;i > 0;i--) {
        pbin[j] = result[i - 1];
        j++;
    }
    pbin[j] = 0;
}
